/** HIGO RICARDO */ 
#include "pico/stdlib.h"

const uint LED_BUILTIN = 25; // Pino do LED integrado

int main() {
    stdio_init_all();
    gpio_init(LED_BUILTIN);
    gpio_set_dir(LED_BUILTIN, GPIO_OUT); // Define o pino como sa√≠da

    while (1) {
        gpio_put(LED_BUILTIN, 1); // LED ligado
        sleep_ms(500);
        gpio_put(LED_BUILTIN, 0); // LED desligado
        sleep_ms(500);
    }
}
