#include <stdio.h>
#include "pico/stdlib.h"
#include "hardware/gpio.h"

#define RED 1
#define YELLOW 5
#define GREEN 9

void setup_leds(void);
void blink_sequence(bool r, bool y, bool g);

int main() {
    //stdio_init_all();
    setup_leds();

    while (true) {
        blink_sequence(true, false, false);
        sleep_ms(4000);

        blink_sequence(false, true, false);
        sleep_ms(2000);

        blink_sequence(false, false, true);
        sleep_ms(4000);
    }

    return 0;
}

void setup_leds(void) {
    gpio_init(RED);
    gpio_set_dir(RED, GPIO_OUT);

    gpio_init(YELLOW);
    gpio_set_dir(YELLOW, GPIO_OUT);

    gpio_init(GREEN);
    gpio_set_dir(GREEN, GPIO_OUT);

    gpio_put(RED, 0);
    gpio_put(YELLOW, 0);
    gpio_put(GREEN, 0);
}

void blink_sequence(bool r, bool y, bool g) {
    gpio_put(RED, r);
    gpio_put(YELLOW, y);
    gpio_put(GREEN, g);
}
